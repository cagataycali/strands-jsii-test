    Quickstart¶

    Get a Strands Agent running in 5 minutes from any language.

    Prerequisites¶

    • Node.js ≥ 20 (jsii runtime requirement)
    • Python ≥ 3.9, Java ≥ 11, Go ≥ 1.21, or .NET ≥ 8.0
    • AWS credentials (for Bedrock) or API keys for other providers

    1. Install¶

    pip install strands-agents-jsii

    <dependency>
    <groupId>io.github.strands-agents</groupId>
    <artifactId>strands-agents-jsii</artifactId>
    <version>0.1.0</version>
</dependency>

    go get github.com/strands-agents/strands-agents-jsii-go

    2. Create an Agent¶

    from strands_agents_jsii import (
    StrandsAgent, AgentConfig,
    BedrockModelProvider, BedrockModelConfig,
)

agent = StrandsAgent(AgentConfig(
    model=BedrockModelProvider(BedrockModelConfig(
        "us.anthropic.claude-sonnet-4-20250514-v1:0"
    )),
    system_prompt="You are a concise assistant."
))

response = agent.invoke("What is the capital of Japan?")
print(response.text)       # "Tokyo"
print(response.input_tokens)   # 42
print(response.output_tokens)  # 5

    import io.github.strandsagents.*;

var agent = new StrandsAgent(new AgentConfig(
    new BedrockModelProvider(new BedrockModelConfig(
        "us.anthropic.claude-sonnet-4-20250514-v1:0",
        "us-west-2", 4096, 0.7, 0.9
    )),
    "You are a concise assistant.",
    null, 50, null, null
));

var response = agent.invoke("What is the capital of Japan?");
System.out.println(response.getText());

    3. Add a Tool¶

    import json
from strands_agents_jsii import ToolDefinition, ToolSpecification

class WeatherTool(ToolDefinition):
    def __init__(self):
        super().__init__(ToolSpecification(
            "get_weather",
            "Get current weather for a city",
            json.dumps({
                "type": "object",
                "properties": {
                    "city": {"type": "string", "description": "City name"}
                },
                "required": ["city"]
            })
        ))

    def execute(self, input_json):
        city = json.loads(input_json)["city"]
        # Your weather API call here
        return json.dumps({"temp": "22°C", "condition": "sunny", "city": city})

agent = StrandsAgent(AgentConfig(
    model=BedrockModelProvider(BedrockModelConfig(
        "us.anthropic.claude-sonnet-4-20250514-v1:0"
    )),
    system_prompt="Use get_weather for weather questions.",
    tools=[WeatherTool()]
))

response = agent.invoke("What's the weather in Tokyo?")
print(response.text)  # "It's 22°C and sunny in Tokyo."

    4. Use Different Providers¶

    # OpenAI
from strands_agents_jsii import OpenAIModelProvider, OpenAIModelConfig
model = OpenAIModelProvider(OpenAIModelConfig("gpt-4o"))

# Anthropic Direct
from strands_agents_jsii import AnthropicModelProvider, AnthropicModelConfig
model = AnthropicModelProvider(AnthropicModelConfig("claude-sonnet-4-20250514"))

# Ollama (Local)
from strands_agents_jsii import OllamaModelProvider, OllamaModelConfig
model = OllamaModelProvider(OllamaModelConfig("qwen3:8b"))

# Google Gemini
from strands_agents_jsii import GeminiModelProvider, GeminiModelConfig
model = GeminiModelProvider(GeminiModelConfig("gemini-2.5-flash"))
